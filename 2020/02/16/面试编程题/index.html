    题目

    * 请用您认为最优化的方式，将arr中的type为4的数据过滤出来，

    ​ * 然后按相同的 name + date（按天）合并value（value累加），

    ​ * 然后按 value 降序(从大到小)排序，

    ​ * 最后每行按照 “${name},${本地日期},售出${sum(value)}部” 的格式，如：”小米2,2017年06月08日,售出5部”, 打印(console.log)出来。

    ​ * 可以使用第三方js库，可以使用es6。

    ​ * 请在半小时内完成。

    ​ 要求最后输出结果样例如下:

    ​ oppor9,2017年06月04日,售出7部

    ​ 魅蓝note2,2017年06月07日,售出6部

    ​ 魅蓝note2,2017年06月02日,售出5部

    ​ 小米2,2017年06月08日,售出5部

    ​ ……

    function printArray(arr) {
      // 通过条件筛选出type为4的值，并将日期做一个截取，只要年月日
      const res = arr.filter(function (item, index, arr) {
        item.date = item.date.split('T')[0]
        // console.log(item.date)
        return item.type === 4
      })
      // console.log(res)
      const newArr = []
      const temp = {}

      // 在for循环里面新定义一个key做name和date的字符串拼接
      // 在判断条件里面，如果在对象temp里面对应的key存在，则将value值累加，
      //否则添加key并将value值新赋值给对应的key
      for (const i in res) {
        const key = `${res[i].name}@${res[i].date}`
        if (temp[key]) {
          temp[key] += res[i].value
        } else {
          temp[key] = res[i].value
        }
      }
      // console.log(temp)

      // 在定义一个for循环，新定义一个临时变量tempObj，遍历对象，对每一个对象的key值做分割，
      // 重新赋值给key，并存储在临时变量tempObj里面，最后将tempObj添加到外面定义的新对象里面
      for (const key in temp) {
        const tempObj = {}
        tempObj.name = key.split('@')[0]
        tempObj.date = key.split('@')[1]
        tempObj.value = temp[key]
        newArr.push(tempObj)
      }
      // console.log(newArr)

      // 定义排序函数
      function compare(prop) {
        return function (a, b) {
          const value1 = a[prop]
          const value2 = b[prop]
          return value2 - value1
        }
      }

      // 调用排序函数，按照value值做降序排序
      newArr.sort(compare('value'))
      // console.log(newArr)

      newArr.forEach((item, index) => {
        const printStr = `${item.name},
            ${item.date.split('-')[0]}年${item.date.split('-')[1]}月
            ${item.date.split('-')[2]}日，售出${item.value}部`
        console.log(printStr)
      })
    }

    printArray(arr);
